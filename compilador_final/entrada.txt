// Arquivo: entrada.txt

// Declaracao de variaveis
VAR REAL salario;
VAR INTEIRO idade;
VAR TEXTO nome;
VAR TEXTO mensagem;

// Interacao com o usuario
ESCREVA("Digite seu nome:");
LEIA_TEXTO(nome);

ESCREVA("Digite sua idade:");
LEIA_INT(idade);

// Estrutura condicional
SE (idade >= 18) {
    ESCREVA("Voce eh maior de idade.");
    salario = 2500.50;
} SENAO {
    ESCREVA("Voce eh menor de idade.");
    salario = 0.0;
}

// Concatenacao de texto e operacoes
mensagem = "Ola " + nome + ", seu salario eh de R$ " + salario;
ESCREVA(mensagem);

// Estrutura de repeticao
VAR INTEIRO contador;
contador = 1;
ESCREVA("Contando ate 3:");
ENQUANTO (contador <= 3) {
    ESCREVA(contador);
    contador = contador + 1;
}

// Vetores (arrays)
VAR REAL notas[4];
ESCREVA("Preenchendo vetor de notas...");
notas[0] = 7.5;
notas[1] = 9.0;
notas[2] = 5.5++;
notas[3] = 8.0;

ESCREVA("A segunda nota foi:");
ESCREVA(notas[1]);

VAR REAL soma;
soma = notas[0] + notas[1] + notas[2] + notas[3];
ESCREVA("A media das notas eh:");
ESCREVA(soma / 4);

ESCREVA("Fim do programa.");
